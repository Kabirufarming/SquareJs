SquareJs OOP

With SquareJs, its posssible to create javascript classes and initialize objects of that class. The initialize method of a class creates an instance of that class.

Example

User = squareJs.Class({
  initialize: function(name, age, weight) {
    this.name = name;
    this.age  = age;
    this.weight = weight;
  }
});

var user = new User('Aminu', '18', '40kg'); //creates a new User object.

You can inherit a class by passing it to squareJs.class

Example

User = squareJs.Class({
  initialize: function(name, age, weight) {
    this.name = name;
    this.age  = age;
    this.weight = weight;
  }
});

SuperUser = turing.Class(User, {
  initialize: function() {
    this.$super('initialize', arguments);
  },

  toString: function() {
    return "SuperUser: " + this.$super('toString');
  }
});

var user = new SuperUser('Aminu', '18', '40kg'); //creates a new SuperUser object which inherits the User class.

Or by using mixins

Example

User = squareJs.Class({
  initialize: function(name, age, weight) {
    this.name = name;
    this.age  = age;
    this.weight = weight;
  }
});

SuperUser = turing.Class({
  include: User,
  initialize: function() {
    this.$super('initialize', arguments);
  },

  toString: function() {
    return "SuperUser: " + this.$super('toString');
  }
});

var user = new SuperUser('Aminu', '18', '40kg'); //creates a new SuperUser object which inherits the User class.
